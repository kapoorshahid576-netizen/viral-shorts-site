// ── CREATE SERIES MODAL ───────────────────────────────────────────────────────
const CreateSeriesModal = ({ onClose, onCreated }) => {
  const [step, setStep] = useState(1);
  const [form, setForm] = useState({
    title: "", topic: "", style: "Listicle", platform: "TikTok", schedule: "Daily", voice: "Male – Deep"
  });
  const [generating, setGenerating] = useState(false);
  const [aiScript, setAiScript] = useState("");
  const [streamText, setStreamText] = useState("");

  const set = (k, v) => setForm(f => ({ ...f, [k]: v }));

  const handleGenerate = async () => {
    setGenerating(true);
    setStep(3);
    setStreamText("");
    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          stream: false,
          messages: [{
            role: "user",
            content: `You are an expert short-form video scriptwriter for viral content.
Generate a compelling ${form.style}-style script for a "${form.title || form.topic}" video for ${form.platform}.
Format:
🎬 HOOK (0–3s): [attention-grabbing opening line]
📋 BODY: [3–5 punchy facts or story beats, each 1–2 sentences]
🔥 CTA (last 3s): [call to action]
Keep it under 60 seconds when spoken aloud. Make it viral-worthy, factual, and engaging.`
          }]
        })
      });
      const data = await res.json();
      const text = data.content?.[0]?.text || "Script generated successfully!";
      setAiScript(text);
      // Simulate streaming by revealing char by char
      let i = 0;
      const interval = setInterval(() => {
        i += 4;
        setStreamText(text.slice(0, i));
        if (i >= text.length) { clearInterval(interval); setGenerating(false); }
      }, 12);
    } catch {
      setAiScript("Error generating script. Please try again.");
      setGenerating(false);
    }
  };

  const handleCreate = () => {
    onCreated({ ...form, id: Date.now(), status: "active", videosPosted: 0, views: "0", thumbnail: "✨", nextPost: "Today 6:00 PM" });
    onClose();
  };

  return (
    <div style={{
      position: "fixed", inset: 0, background: "rgba(0,0,0,.7)", zIndex: 1000,
      display: "flex", alignItems: "center", justifyContent: "center", padding: 20
    }}>
      <div style={{
        background: CARD, border: `1px solid ${BORDER}`, borderRadius: 20,
        width: "100%", maxWidth: 560, maxHeight: "90vh", overflow: "auto",
        padding: 32, position: "relative"
      }}>
        <button onClick={onClose} style={{
          position: "absolute", top: 16, right: 16, background: "transparent",
          border: "none", color: MUTED, fontSize: 22, cursor: "pointer"
        }}>✕</button>

        {/* Step indicator */}
        <div style={{ display: "flex", gap: 8, marginBottom: 28 }}>
          {["Series Setup", "Schedule", "Preview Script"].map((s, i) => (
            <div key={i} style={{ flex: 1, textAlign: "center" }}>
              <div style={{
                height: 3, borderRadius: 2, marginBottom: 6,
                background: step > i ? ACCENT : step === i + 1 ? ACCENT : BORDER
              }} />
              <span style={{ fontSize: 11, color: step === i + 1 ? ACCENT : MUTED }}>{s}</span>
            </div>
          ))}
        </div>

        {step === 1 && (
          <div style={{ display: "flex", flexDirection: "column", gap: 18 }}>
            <h2 style={{ color: TEXT, fontSize: 22, fontWeight: 800, margin: 0 }}>Create New Series</h2>
            <Field label="Series Title" value={form.title} onChange={v => set("title", v)} placeholder="e.g. Mind-Blowing Science Facts" />
            <Field label="Topic / Niche" value={form.topic} onChange={v => set("topic", v)} placeholder="e.g. Science, History, Finance…" />
            <SelectField label="Video Style" value={form.style} options={["Listicle", "Narrated Story", "ASMR Cutting", "Tutorial", "News-style"]} onChange={v => set("style", v)} />
            <SelectField label="Platform" value={form.platform} options={["TikTok", "YouTube Shorts", "Instagram Reels", "All Platforms"]} onChange={v => set("platform", v)} />
            <SelectField label="AI Voice" value={form.voice} options={["Male – Deep", "Male – Energetic", "Female – Calm", "Female – Upbeat"]} onChange={v => set("voice", v)} />
            <button onClick={() => setStep(2)} style={primaryBtn}>Continue →</button>
          </div>
        )}

        {step === 2 && (
          <div style={{ display: "flex", flexDirection: "column", gap: 18 }}>
            <h2 style={{ color: TEXT, fontSize: 22, fontWeight: 800, margin: 0 }}>Posting Schedule</h2>
            <div>
              <div style={{ color: TEXT, fontSize: 13, marginBottom: 10 }}>Frequency</div>
              <div style={{ display: "flex", flexWrap: "wrap", gap: 8 }}>
                {["Daily", "2x/week", "3x/week", "5x/week", "Custom"].map(s => (
                  <Pill key={s} label={s} active={form.schedule === s} onClick={() => set("schedule", s)} />
                ))}
              </div>
            </div>
            <Field label="Best Posting Time" value="6:00 PM" onChange={() => {}} placeholder="6:00 PM" />
            <div style={{ background: CARD2, border: `1px solid ${BORDER}`, borderRadius: 10, padding: 14 }}>

              <div style={{ color: MUTED, fontSize: 13 }}>ViralShorts will analyze your audience and auto-adjust posting times for maximum reach.</div>
            </div>
            <div style={{ display: "flex", gap: 10 }}>
              <button onClick={() => setStep(1)} style={secondaryBtn}>← Back</button>
              <button onClick={handleGenerate} style={{ ...primaryBtn, flex: 1 }}>Generate Preview Script</button>
            </div>
          </div>
        )}

        {step === 3 && (
          <div style={{ display: "flex", flexDirection: "column", gap: 18 }}>
            <h2 style={{ color: TEXT, fontSize: 22, fontWeight: 800, margin: 0 }}>AI-Generated Script Preview</h2>
            <div style={{
              background: CARD2, border: `1px solid ${BORDER}`, borderRadius: 12,
              padding: 18, minHeight: 220, fontFamily: "monospace", fontSize: 13,

            {!generating && (
              <div style={{ display: "flex", gap: 10 }}>
                <button onClick={handleGenerate} style={secondaryBtn}>↺ Regenerate</button>
                <button onClick={handleCreate} style={{ ...primaryBtn, flex: 1 }}>🚀 Launch Series</button>
              </div>
            )}
          </div>
        )}
      </div>
    </div>
  );
};

      </div>`;

  grid.innerHTML = html;
}

  const max = Math.max(...BARS.map(b => b.views));
  document.getElementById('bar-chart').innerHTML = BARS.map(b => `
    <div class="bar-col">
      <div class="bar-label-top">${(b.views/1000).toFixed(0)}K</div>
      <div class="bar-fill" style="height:${(b.views/max)*120}px"></div>
      <div class="bar-label-bot">${b.day}</div>
    </div>
  `).join('');
 
  const widths = { "1.2M":"100%","880K":"73%","430K":"36%","190K":"16%" };
  document.getElementById('series-perf').innerHTML = SERIES.map(s => `
    <div class="perf-row">
      <div class="perf-emoji">${s.thumbnail}</div>
      <div class="perf-info">
        <div class="perf-top">
          <span class="perf-name">${s.title}</span>
          <span class="perf-views">${s.views}</span>
        </div>
        <div class="perf-bar-bg">
          <div class="perf-bar-fill" style="width:${widths[s.views]||'10%'}"></div>
        </div>
      </div>
    </div>
  `).join('');
}

async function fetchInsight() {
  const btn = document.getElementById('insight-btn');
  const placeholder = document.getElementById('insight-placeholder');
  const textEl = document.getElementById('insight-text');
  btn.disabled = true;
  btn.textContent = 'Analyzing…';
  placeholder.style.display = 'none';
  textEl.style.display = 'block';
  textEl.textContent = '';
 
  const activeTitles = SERIES.filter(s => s.status === 'active').map(s => s.title).join(', ');
  try {
    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        messages: [{
          role: 'user',
          content: `You are a viral content strategist. Based on these active video series: ${activeTitles} — give 3 short, punchy, actionable tips (1 sentence each) to maximize views this week on TikTok and YouTube Shorts. Start each tip with a relevant emoji. Be specific and bold.`
        }]
      })
    });
    const data = await res.json();
    const text = data.content?.[0]?.text || 'Post consistently for best results!';
    // Typewriter
    let i = 0;
    const iv = setInterval(() => {
      i += 3;
      textEl.textContent = text.slice(0, i);
      if (i >= text.length) clearInterval(iv);
    }, 10);
  } catch {
    textEl.textContent = '💡 Post at 6–9 PM on weekdays for the highest engagement.\n🔥 Use trending sounds in your first 3 seconds to stop the scroll.\n📈 Respond to every comment in the first hour after posting to boost the algorithm.';
  }
  btn.disabled = false;
  btn.textContent = 'Refresh';
}

───────────────────────────────────────────────────────────── */
let modalCurrentStep = 1;
let selectedSchedule = 'Daily';
 
function openModal() {
  modalCurrentStep = 1;
  showModalStep(1);
  document.getElementById('m-title').value = '';
  document.getElementById('m-topic').value = '';
  document.getElementById('modal-overlay').classList.add('open');
}
function closeModal() {
  document.getElementById('modal-overlay').classList.remove('open');
}
 
document.getElementById('modal-overlay').addEventListener('click', function(e) {
  if (e.target === this) closeModal();
});
 
function showModalStep(n) {
  [1,2,3].forEach(i => {
    document.getElementById('modal-step-' + i).style.display = i === n ? 'block' : 'none';
    const line = document.getElementById('step-line-' + i);
    const name = document.getElementById('step-name-' + i);
    if (i < n) { line.className = 'step-line done'; name.className = 'step-name'; }
    else if (i === n) { line.className = 'step-line active'; name.className = 'step-name active'; }
    else { line.className = 'step-line'; name.className = 'step-name'; }
  });
  modalCurrentStep = n;
}
 
function modalStep(n) { showModalStep(n); }
 
function setSchedule(val, btn) {
  selectedSchedule = val;
  document.querySelectorAll('#schedule-pills .pill').forEach(p => p.classList.remove('active'));
  btn.classList.add('active');
}
 
async function generateScript() {
  showModalStep(3);
  const scriptBox = document.getElementById('script-box');
  const actions = document.getElementById('script-actions');
  scriptBox.innerHTML = '<span style="color:#7070a0;animation:pulse 1.2s infinite">Generating script with AI…</span>';
  actions.style.display = 'none';
 
  const title = document.getElementById('m-title').value;
  const topic = document.getElementById('m-topic').value;
  const style = document.getElementById('m-style').value;
  const platform = document.getElementById('m-platform').value;
 
  try {
    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        messages: [{
          role: 'user',
          content: `You are an expert short-form video scriptwriter for viral content.
Generate a compelling ${style}-style script for a "${title || topic || 'viral content'}" video for ${platform}.
Format exactly like this:
🎬 HOOK (0–3s): [attention-grabbing opening line]
 
📋 BODY:
• [fact or story beat 1]
• [fact or story beat 2]
• [fact or story beat 3]
• [fact or story beat 4]
• [fact or story beat 5]
 
🔥 CTA (last 3s): [call to action]
 
Keep it under 60 seconds when spoken aloud. Make it viral-worthy, factual, and engaging.`
        }]
      })
    });
    const data = await res.json();
    const text = data.content?.[0]?.text || 'Script generated!';
    scriptBox.textContent = '';
    let i = 0;
    const iv = setInterval(() => {
      i += 4;
      scriptBox.textContent = text.slice(0, i);
      if (i >= text.length) {
        clearInterval(iv);
        actions.style.display = 'flex';
      }
    }, 12);
  } catch {
    scriptBox.textContent = `🎬 HOOK (0–3s): You won't believe what scientists just discovered…\n\n📋 BODY:\n• Quantum entanglement allows particles to communicate faster than light.\n• A day on Venus is longer than a year on Venus.\n• There are more trees on Earth than stars in the Milky Way.\n• The human brain generates 23 watts of power — enough to light a light bulb.\n• Honey never expires — 3,000-year-old honey found in Egyptian tombs was still edible.\n\n🔥 CTA (last 3s): Follow for more mind-blowing facts every day!`;
    actions.style.display = 'flex';
  }
}
 
function launchSeries() {
  const title = document.getElementById('m-title').value || document.getElementById('m-topic').value || 'New Series';
  const platform = document.getElementById('m-platform').value;
  const style = document.getElementById('m-style').value;
  const emojis = ['✨','🌟','💥','🔥','⚡','🎯','🚀'];
  SERIES.unshift({
    id: Date.now(), title, topic: document.getElementById('m-topic').value,
    platform, status: 'active', nextPost: 'Today 6:00 PM', videosPosted: 0,
    views: '0', style, schedule: selectedSchedule,
    thumbnail: emojis[Math.floor(Math.random()*emojis.length)]
  });
  closeModal();
  renderDashboard();
  renderSeriesGrid();
  showToast('🚀 Series launched! First video is being generated.');
}
 
/* ─────────────────────────────────────────────────────────────
   TOAST
───────────────────────────────────────────────────────────── */
function showToast(msg) {
  const t = document.createElement('div');
  t.textContent = msg;
  Object.assign(t.style, {
    position:'fixed', bottom:'28px', left:'50%', transform:'translateX(-50%)',
    background:'linear-gradient(135deg,#a259ff,#ff5eaa)', color:'#fff',
    padding:'12px 24px', borderRadius:'10px', fontSize:'14px', fontWeight:'700',
    zIndex:'9999', animation:'slideIn .25s ease', boxShadow:'0 8px 32px rgba(162,89,255,.4)',
    whiteSpace:'nowrap'
  });
  document.body.appendChild(t);
  setTimeout(() => t.remove(), 3500);
}
